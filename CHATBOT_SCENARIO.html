import os
from openai import OpenAI

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def get_ai_response(user_message):
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": "당신은 친절한 AI 도우미입니다."},
            {"role": "user", "content": user_message}
        ]
    )
    return response.choices[0].message.content

static/script.js

async function sendMessage() {
    const input = document.getElementById("user-input");
    const message = input.value.trim();
    if (!message) return;

    addMessage("You", message);
    input.value = "";

    const res = await fetch("/chat", {
        method: "POST",
        headers: {"Content-Type": "application/json"},
        body: JSON.stringify({ message })
    });
    const data = await res.json();
    addMessage("Bot", data.reply);
}

function addMessage(sender, text) {
    const div = document.createElement("div");
    div.className = `message ${sender.toLowerCase()}`;
    div.innerHTML = `<strong>${sender}:</strong> ${text}`;
    document.getElementById("messages").appendChild(div);
}

import json, os

HISTORY_FILE = "chat_history.json"

def save_message(role, content):
    history = load_history()
    history.append({"role": role, "content": content})
    with open(HISTORY_FILE, "w") as f:
        json.dump(history, f, ensure_ascii=False, indent=2)

def load_history():
    if os.path.exists(HISTORY_FILE):
        with open(HISTORY_FILE) as f:
            return json.load(f)
    return []

최종 app.py

from flask import Flask, render_template, request, jsonify
from dotenv import load_dotenv
from chat_service import get_ai_response
from history import save_message, load_history

load_dotenv()
app = Flask(__name__)

@app.route("/")
def index():
    return render_template("index.html")

@app.route("/chat", methods=["POST"])
def chat():
    user_message = request.json.get("message")
    if not user_message:
        return jsonify({"error": "메시지가 비어있습니다"}), 400
    try:
        save_message("user", user_message)
        reply = get_ai_response(user_message)
        save_message("assistant", reply)
        return jsonify({"reply": reply})
    except Exception as e:
        return jsonify({"error": str(e)}), 500

@app.route("/history")
def get_history():
    return jsonify(load_history())

if __name__ == "__main__":
    app.run(debug=True)

원칙 1. "파일이 안 겹치면 충돌은 안 난다"

작업을 나눌 때 "기능"이 아니라 "어떤 파일을 건드리는지"까지 명시해야 합니다.

→ 충돌!
✅
파일 경계까지 나눈 경우

지연: 회원가입 → auth/register.py

→ 파일이 안 겹침
→ 충돌 없음!

원칙 2. 충돌이 안 나는 프로젝트 구조로 설계

TASK.md를 잘 나누려면, 그 전에 프로젝트 구조 자체를 충돌이 안 나게 설계해야 합니다.

❌ 나쁜 구조 — 모든 게 하나에
project/ ├── app.py # 7명 전원이 수정 ├── service.py # 다들 건드림 ├── models.py # 다들 건드림 └── utils.py # 다들 건드림 → 충돌 지옥 🔥
✅ 좋은 구조 — 기능별 파일 분리
